public class Test {
    public static void main(String[] args) {
        Node A = new Node("A");
        Node B = new Node("B");
        Node C = new Node("C");
        Node D = new Node("D");
        Node E = new Node("E");
        Node F = new Node("F");

        A.left = B;
        A.right = C;
        B.left = D;
        C.left = E;
        C.right = F;


        
    }

    
    static void preOrder(Node node){
        if(node==null) return;
        Stack<Node> stack = new Stack<>();
        stack.push(node);
        while(!stack.isEmpty()){
            Node index = stack.pop();
            System.out.print(index.value);
            if(index.right!=null) stack.push(index.right);
            if(index.left!=null) stack.push(index.left);
        }
    }

    static  void postOrder(Node node){
        if(node==null) return;
        Stack<Node> stack = new Stack<>();
        Stack<Node> stacklast = new Stack<>();
        stack.push(node);
        while(!stack.isEmpty()){
            Node index = stack.pop();
            stacklast.push(index);
            if(index.left!=null) stack.push(index.left);
            if(index.right!=null) stack.push(index.right);
        }

        while (!stacklast.isEmpty()){
            System.out.print(stacklast.pop().value);
        }

    }




    static void nicePrint(Node node){
        Queue<Node> queue = new LinkedList<>();
        queue.add(node);
        List<List<Node>> res = new ArrayList<>();
        while(!queue.isEmpty()){
            List<Node> list = new ArrayList<>() ;
            int size = queue.size();
            for (int i = 0; i < size; i++) {
                list.add(queue.poll());
            }
            for(Node index:list){
                if(index.left != null) queue.add(index.left);
                if(index.right != null) queue.add(index.right);
            }
            res.add(list);
        }

        for (List<Node> index:res){
            for (Node n:index){
                System.out.print(n.value);
            }
            System.out.println();
        }
    }



}

class Node{
    String value;
    Node left;
    Node right;

    public Node(String value){
        this.value = value;
    }
}
